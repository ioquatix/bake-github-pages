def build_gem(root: "pkg")
	# Ensure the output directory exists:
	FileUtils.mkdir_p("pkg")
	
	output_path = File.join('pkg', @gemspec.file_name)
	
	::Gem::Package.build(@gemspec, false, false, output_path)
